比如二叉树，用OCaml，可以这么定义

type 'a tree = | Empty | Tree of 'a tree * 'a * 'a tree;;

检查其中是否存在某个元素

let rec member elem tree = match tree with | Empty -> false | Tree(left, node, right) -> match Elem.compare elem node with | Equal -> true | Less -> member elem left | Greater -> member elem right;;

遍历

let rec iter tree = match tree with | Empty -> empty | Tree(left, node, right) -> chain (iter left) (cons node (iter right));;

Iterator相关定义

type 'a stream = | Nil | Cons of 'a * 'a iterator and 'a iterator = unit -> 'a stream;; let empty () = Nil;; let cons elem iterator () = Cons(elem, iterator);; let rec chain a b () = match a () with | Nil -> b () | Cons(h, t) -> Cons(h, (chain t b));;
